SELECT I.SUMMARY, I.DESCRIPTION, I2.NAME
FROM ISSUES I
JOIN ISSUESLISTS I2 on I2.ID = I.ISSUESLIST_ID
ORDER BY I.ID;

SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY I.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY U.ID;
